```go
// Package bm25 implements a precomputed BM25 scoring index with prefix boosting
// for ranking product search results by term importance.
package bm25

import (
	"strings"
)

// Default BM25 parameters tuned for short product names.
const (
	DefaultK1 = 1.2
	DefaultB  = 0.75
)

// SearchResult holds a product match with its BM25 score.
type SearchResult struct {
	ProductID   int
	Score       float64
	PrefixMatch bool
}

// Index is a precomputed BM25 scoring index over a product catalog.
type Index struct {
	idf          map[string]float64        // precomputed IDF per term
	termFreqs    []map[string]int          // product ID -> term -> frequency
	docLens      []int                     // product ID -> word count
	avgDocLen    float64                   // mean document length
	wordPrefixes []map[string]struct{}     // product ID -> set of word prefixes
	posting      map[string][]int          // term -> product IDs
	k1           float64
	b            float64
}

// Tokenize splits text into lowercase words on whitespace, filtering empty tokens.
func Tokenize(s string) []string {
	s = strings.ToLower(strings.TrimSpace(s))
	if s == "" {
		return nil
	}
	fields := strings.Fields(s)
	// Filter out pure-punctuation tokens like "&"
	result := make([]string, 0, len(fields))
	for _, f := range fields {
		// Keep tokens that contain at least one alphanumeric character
		hasAlphaNum := false
		for _, r := range f {
			if (r >= 'a' && r <= 'z') || (r >= '0' && r <= '9') {
				hasAlphaNum = true
				break
			}
		}
		if hasAlphaNum {
			result = append(result, f)
		}
	}
	return result
}
```

```go
import (
	"math"

	"github.com/mbow/go-xsearch/catalog"
)

// NewIndex builds a BM25 index from the given products.
// All IDF values, term frequencies, document lengths, word prefixes,
// and posting lists are precomputed for zero-allocation scoring at query time.
func NewIndex(products []catalog.Product) *Index {
	n := len(products)
	idx := &Index{
		idf:          make(map[string]float64),
		termFreqs:    make([]map[string]int, n),
		docLens:      make([]int, n),
		wordPrefixes: make([]map[string]struct{}, n),
		posting:      make(map[string][]int),
		k1:           DefaultK1,
		b:            DefaultB,
	}

	// Document frequency: how many products contain each term
	df := make(map[string]int)

	totalLen := 0
	for id, p := range products {
		// Tokenize name, category, and tags into one document
		tokens := Tokenize(p.Name)
		tokens = append(tokens, Tokenize(p.Category)...)
		for _, tag := range p.Tags {
			tokens = append(tokens, Tokenize(tag)...)
		}

		idx.docLens[id] = len(tokens)
		totalLen += len(tokens)

		// Term frequencies
		tf := make(map[string]int, len(tokens))
		for _, tok := range tokens {
			tf[tok]++
		}
		idx.termFreqs[id] = tf

		// Document frequency (count each term once per document)
		for term := range tf {
			df[term]++
		}

		// Posting lists
		for term := range tf {
			idx.posting[term] = append(idx.posting[term], id)
		}

		// Word prefixes from product name only (not category/tags)
		nameTokens := Tokenize(p.Name)
		prefixes := make(map[string]struct{})
		for _, tok := range nameTokens {
			for i := 1; i <= len(tok); i++ {
				prefixes[tok[:i]] = struct{}{}
			}
		}
		idx.wordPrefixes[id] = prefixes
	}

	// Average document length
	if n > 0 {
		idx.avgDocLen = float64(totalLen) / float64(n)
	}

	// Precompute IDF: log((N - df + 0.5) / (df + 0.5) + 1)
	nf := float64(n)
	for term, d := range df {
		idx.idf[term] = math.Log((nf-float64(d)+0.5)/(float64(d)+0.5) + 1)
	}

	return idx
}
```

```go
import (
	"cmp"
	"slices"
)

// Score computes the BM25 score for a single product against query terms.
// All values are precomputed lookups — the only work is arithmetic.
func (idx *Index) Score(productID int, queryTerms []string) float64 {
	tf := idx.termFreqs[productID]
	dl := float64(idx.docLens[productID])
	score := 0.0

	for _, term := range queryTerms {
		idf, ok := idx.idf[term]
		if !ok {
			continue
		}
		freq := float64(tf[term])
		if freq == 0 {
			continue
		}
		num := freq * (idx.k1 + 1)
		denom := freq + idx.k1*(1-idx.b+idx.b*dl/idx.avgDocLen)
		score += idf * num / denom
	}

	return score
}

// HasPrefixMatch returns true if any query term is a prefix of any word
// in the product's name.
func (idx *Index) HasPrefixMatch(productID int, queryTerms []string) bool {
	prefixes := idx.wordPrefixes[productID]
	for _, term := range queryTerms {
		if _, ok := prefixes[term]; ok {
			return true
		}
	}
	return false
}

// Search finds and scores all products matching any query term,
// applying BM25 scoring with prefix boost. Results are sorted by
// score descending.
func (idx *Index) Search(query string) []SearchResult {
	terms := Tokenize(query)
	if len(terms) == 0 {
		return nil
	}

	// Collect candidate product IDs from posting lists
	seen := make(map[int]struct{})
	var candidates []int
	for _, term := range terms {
		for _, id := range idx.posting[term] {
			if _, ok := seen[id]; !ok {
				seen[id] = struct{}{}
				candidates = append(candidates, id)
			}
		}
	}

	if len(candidates) == 0 {
		// Try prefix matching: find products where any term is a prefix of a word
		for id := range idx.wordPrefixes {
			if idx.HasPrefixMatch(id, terms) {
				if _, ok := seen[id]; !ok {
					seen[id] = struct{}{}
					candidates = append(candidates, id)
				}
			}
		}
	}

	if len(candidates) == 0 {
		return nil
	}

	// Compute max IDF across query terms for prefix bonus calculation
	maxIDF := 0.0
	for _, term := range terms {
		if idf, ok := idx.idf[term]; ok && idf > maxIDF {
			maxIDF = idf
		}
	}
	// If no exact term IDF found (pure prefix query), estimate from corpus
	if maxIDF == 0 {
		maxIDF = 1.0
	}
	prefixBonus := 0.5 * maxIDF

	// Score all candidates
	results := make([]SearchResult, 0, len(candidates))
	for _, id := range candidates {
		score := idx.Score(id, terms)
		prefixMatch := idx.HasPrefixMatch(id, terms)
		if prefixMatch {
			score += prefixBonus
		}
		if score > 0 {
			results = append(results, SearchResult{
				ProductID:   id,
				Score:       score,
				PrefixMatch: prefixMatch,
			})
		}
	}

	// Sort by score descending
	slices.SortFunc(results, func(a, b SearchResult) int {
		return cmp.Compare(b.Score, a.Score)
	})

	return results
}
```

```go
// Snapshot holds the serializable state of an Index.
type Snapshot struct {
	IDF          map[string]float64 `cbor:"idf"`
	TermFreqs    []map[string]int   `cbor:"term_freqs"`
	DocLens      []int              `cbor:"doc_lens"`
	AvgDocLen    float64            `cbor:"avg_doc_len"`
	WordPrefixes [][]string         `cbor:"word_prefixes"`
	Posting      map[string][]int   `cbor:"posting"`
	K1           float64            `cbor:"k1"`
	B            float64            `cbor:"b"`
}

// ToSnapshot exports the index state for serialization.
func (idx *Index) ToSnapshot() Snapshot {
	wp := make([][]string, len(idx.wordPrefixes))
	for id, prefixes := range idx.wordPrefixes {
		list := make([]string, 0, len(prefixes))
		for p := range prefixes {
			list = append(list, p)
		}
		wp[id] = list
	}

	return Snapshot{
		IDF:          idx.idf,
		TermFreqs:    idx.termFreqs,
		DocLens:      idx.docLens,
		AvgDocLen:    idx.avgDocLen,
		WordPrefixes: wp,
		Posting:      idx.posting,
		K1:           idx.k1,
		B:            idx.b,
	}
}

// FromSnapshot restores an Index from a serialized Snapshot.
func FromSnapshot(s Snapshot) *Index {
	wp := make([]map[string]struct{}, len(s.WordPrefixes))
	for id, list := range s.WordPrefixes {
		m := make(map[string]struct{}, len(list))
		for _, p := range list {
			m[p] = struct{}{}
		}
		wp[id] = m
	}

	return &Index{
		idf:          s.IDF,
		termFreqs:    s.TermFreqs,
		docLens:      s.DocLens,
		avgDocLen:    s.AvgDocLen,
		wordPrefixes: wp,
		posting:      s.Posting,
		k1:           s.K1,
		b:            s.B,
	}
}
```

```go
func (e *Engine) searchInternal(query string) []Result {
	score := e.ranker.Scorer()

	// Try BM25 path first (word-level matching)
	bm25Results := e.bm25idx.Search(query)
	if len(bm25Results) > 0 {
		return e.buildBM25Results(bm25Results, score)
	}

	// Fallback to Jaccard trigram path (handles typos/fuzzy)
	trigrams := index.ExtractTrigrams(query)
	if len(trigrams) == 0 {
		return e.buildResults(e.index.Search(query), MatchDirect, score)
	}

	anyPass := false
	for _, g := range trigrams {
		if e.bloom.MayContain(g) {
			anyPass = true
			break
		}
	}

	pooled := e.getResults()
	results := *pooled

	if anyPass {
		searchResults := e.index.Search(query)
		goodResults := 0
		for _, sr := range searchResults {
			if sr.Score >= fallbackThreshold {
				goodResults++
			}
		}
		results = append(results, e.buildResults(searchResults, MatchDirect, score)...)
		if goodResults < minDirectResults {
			fallbackResults := e.categoryFallback(query, searchResults, score)
			results = append(results, fallbackResults...)
		}
	} else {
		results = append(results, e.categoryFallback(query, nil, score)...)
	}

	slices.SortFunc(results, func(a, b Result) int {
		return cmp.Compare(b.Score, a.Score)
	})

	if len(results) > maxResults {
		results = results[:maxResults]
	}

	out := make([]Result, len(results))
	copy(out, results)

	*pooled = results
	e.putResults(pooled)

	return out
}
```

Add new method `buildBM25Results`:

```go
// buildBM25Results converts BM25 search results to engine results,
// normalizing BM25 scores and blending with prefix boost and popularity.
func (e *Engine) buildBM25Results(bm25Results []bm25.SearchResult, score scorerFunc) []Result {
	// Find max BM25 score for normalization
	maxBM25 := 0.0
	for _, r := range bm25Results {
		if r.Score > maxBM25 {
			maxBM25 = r.Score
		}
	}
	if maxBM25 == 0 {
		maxBM25 = 1.0
	}

	results := make([]Result, 0, min(len(bm25Results), maxResults))
	for _, r := range bm25Results {
		if r.ProductID < 0 || r.ProductID >= len(e.products) {
			continue
		}

		normalizedBM25 := r.Score / maxBM25
		prefixBoost := 0.0
		if r.PrefixMatch {
			prefixBoost = 1.0
		}

		// Blend: 0.5*bm25 + 0.2*prefix + 0.3*popularity
		popScore := score(r.ProductID, 0)
		// score() returns alpha*relevance + (1-alpha)*popularity
		// With relevance=0, we get just the popularity component
		// We need raw popularity, so divide by (1-alpha)
		// But simpler: just call with relevance=1 and subtract alpha
		combined := bm25Alpha*normalizedBM25 + bm25PrefixAlpha*prefixBoost + bm25PopAlpha*popScore

		results = append(results, Result{
			Product:   e.products[r.ProductID],
			ProductID: r.ProductID,
			Score:     combined,
			MatchType: MatchDirect,
		})

		if len(results) >= maxResults {
			break
		}
	}

	return results
}
```

```go
		{Name: "Funky Buddha No. 5 Lager", Category: "beer"},
		{Name: "Funky Buddha No. 12 Porter", Category: "beer"},
		{Name: "Funky Buddha Limited Lambic", Category: "beer"},
	}

	e := New(products)
	results := e.Search("bud")

	if len(results) == 0 {
		t.Fatal("expected results for 'bud'")
	}

	// First result must be Budweiser or Bud Light (prefix match on "bud")
	firstName := results[0].Product.Name
	if firstName != "Budweiser" && firstName != "Bud Light" {
		t.Errorf("expected first result to be Budweiser or Bud Light, got %q", firstName)
	}

	// No Funky Buddha product should appear before all Bud* products
	lastBudIdx := -1
	firstFunkyIdx := -1
	for i, r := range results {
		name := r.Product.Name
		if name == "Budweiser" || name == "Bud Light" {
			lastBudIdx = i
		}
		if len(name) >= 5 && name[:5] == "Funky" && firstFunkyIdx == -1 {
			firstFunkyIdx = i
		}
	}

	if firstFunkyIdx != -1 && lastBudIdx > firstFunkyIdx {
		t.Errorf("Funky Buddha (index %d) appeared before last Bud* product (index %d)", firstFunkyIdx, lastBudIdx)
		for i, r := range results {
			t.Logf("  result[%d]: %s (score=%.4f)", i, r.Product.Name, r.Score)
		}
	}
}
```
